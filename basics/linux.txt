--help описание комманды 
cd test  - переход в папку test (если это папка и она существует)
cd ..    - переход в родительскую папку 
cd .     - переход в эту же папку (мы никуда не перейдём)
cd       - переход в домашнюю папку cd ~
pwd      - показывает абсолютный путь до текущей папки 
mkdir test - создание новой папки test 
mkdir test/new_dir -  создание новой папки в уже существуещей папке
touch test.txt     -  создание нового файла test.txt
rm test.txt        -  удаление файла test.txt 
rm -rf test        -  удаление папки test со всем её содержимым 
sudo rm -rf /      - удаление всех данных (не использовать!!!)
cat test.txt       - просмотр содержимого файла 
ls test            - просмотр содержимого папки 
ls test -a         - просмотр всего содержимого папки (включая скрытых)
ls l               - просмотр файлов с более подробной информацией 

nano test.txt      - открыть файл в редакторе nano
ctrl+s сохранить 
ctrl+x выйти
